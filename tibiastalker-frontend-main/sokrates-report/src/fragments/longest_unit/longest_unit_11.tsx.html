  function handleSubmit(e?: FormEvent) {
    e?.preventDefault?.();
    const brut = q;

    const parsed = rechercheSchema.safeParse(brut);

    if (!parsed.success) {
      const msg =
        parsed.error.issues[0]?.message ||
        "La requête de recherche est invalide.";
      setErreur(msg);
      return;
    }

    const valeurValide = parsed.data;
    setErreur("");
    onSubmit?.(valeurValide);
  }
