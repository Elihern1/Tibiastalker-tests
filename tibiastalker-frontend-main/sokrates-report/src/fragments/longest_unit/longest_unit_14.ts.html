export async function startConnection() {
  if (connection.state === signalR.HubConnectionState.Connected) {
    return;
  }

  try {
    await connection.start();
    console.log("SignalR Connected.");
  } catch (err) {
    console.error("Error starting SignalR:", err);
  }
}
